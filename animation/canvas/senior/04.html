四、高级动画

4.1 缓动动画

缓动动画，指带有一定缓冲的动画，物体在一定时间内渐进加速或者减速，从而使动画更加的真实和自然。

缓动动画分为两种，缓入和缓出。例如：汽车起步逐渐加速 —— 缓入动画；汽车停车时逐渐减速 —— 缓出动画；

制作缓动动画共五步：

  1. 定义 0-1 的缓动系数 easing
  2. 计算物体距离终点的距离
  3. 计算当前的速度：距离 X 缓动系数
  4. 计算新的位置：当前位置 + 当前速度
  5. 重复执行 2 - 4 步，直到达到目标
  • 语法：
const targerX = 任意位置
const targetY = 任意位置
// 动画循环
const vx = (targetX - object.x) * easing
const vy = (targetY - object.y) * easing

4.1.1 实现X轴或Y轴的缓动动画

  • 代码
<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8" />
        <title>Document</title>
    </head>
    <body>
        <canvas id="myCanvas" width="600" height="400" style="background: #ccc"> </canvas>

        <script>
            class Ball{
                constructor(x = 0,y = 0,radius = 12,color = '#6699FF'){
                    //小球中心的x坐标
                    this.x = x;
                     //小球中心的y坐标
                    this.y = y;
                    //小球半径
                    this.radius = radius;
                    // 小球颜色
                    this.color = color;
                }

                fill (ctx){
                    ctx.fillStyle = this.color;
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.radius, 0, 360 * Math.PI / 180);
                    ctx.closePath();
                    ctx.fill()
                }
            }


            // 获取canvas对象
            const canvas = document.getElementById("myCanvas");
            // 获取上下文对象context
            const ctx = canvas.getContext("2d");

           // 实例化一个小球
            const ball = new Ball(0, canvas.height / 2,30);

            // 定义终点坐标
            const targetX = canvas.width * (3 / 4);
            // 定义缓动系数
            const easing = 0.02;

            (function drawFrame() {
                window.requestAnimationFrame(drawFrame, canvas);
                // 清空画布
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                const vx = (targetX - ball.x) * easing
                ball.x += vx

                ball.fill(ctx)

                if(vx < 0.001){
                    ball.x = 0;
                }

            })();
        </script>
    </body>
</html>

由实验可知，小球的速度是一个由快到慢的过程，原理很简单，距离越短，距离与方向的乘积越小，那么方向速度就越来越小。

缓动系数使 0-1的数，越接近1，小球移动速度越快。

因为缓动动画的每一帧的速度都不同，所以它更加的自然，所以更受到开发者的青睐。

<html lang="en">

            // 定义鼠标事件
            const getMouse = (element) => {

                    mouse.x = x - element.offsetLeft
                    mouse.y = y - element.offsetTop
                })

            }

            const canvas = document.getElementById("myCanvas");
            // 获取上下文对象context
            const ctx = canvas.getContext("2d");

            const ball = new Ball(canvas.width / 2, canvas.height / 2,30);
            const smallBall = new Ball(canvas.width / 2, canvas.height / 2, 25, 'red')

            const mouse = getMouse(canvas);
            // 定义缓动系数
            const easing = 0.03;

            (function drawFrame() {
                window.requestAnimationFrame(drawFrame, canvas);
                // 清空画布
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                const vx = (mouse.x - ball.x) * easing
                const vy = (mouse.y - ball.y) * easing

                ball.x += vx
                ball.y += vy

                ball.fill(ctx)

                const vx2 = (ball.x - smallBall.x) * easing
                const vy2 = (ball.y - smallBall.y) * easing

                smallBall.x += vx2
                smallBall.y += vy2

                smallBall.fill(ctx)

            })();
        </script>
    </body>

4.1.4 处理半径、透明度、颜色

  • 代码
<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8" />
        <title>Document</title>
    </head>
    <body>
        <canvas id="myCanvas" width="600" height="400" style="background: #ccc"> </canvas>

        <script>
            class Ball{
                constructor(x = 0,y = 0,radius = 12,color = '#6699FF'){
                    //小球中心的x坐标
                    this.x = x;
                     //小球中心的y坐标
                    this.y = y;
                    //小球半径
                    this.radius = radius;
                    // 小球颜色
                    this.color = color;
                }

                fill (ctx){
                    ctx.fillStyle = this.color;
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.radius, 0, 360 * Math.PI / 180);
                    ctx.closePath();
                    ctx.fill()
                }
            }

            // 获取canvas对象
            const canvas = document.getElementById("myCanvas");
            // 获取上下文对象context
            const ctx = canvas.getContext("2d");

            // ======== 处理小球半径 =======
            let ball1 = new Ball(canvas.width / 4, canvas.height / 2);
            let minRadius = 5;  //最小半径
            let maxRadius = 50; //最大半径
            let state1 = 1; // 0缩小 1放大
            let vRadius = 0;

            // ======== 处理小球透明度 =======
            let ball2 = new Ball(canvas.width / 2, canvas.height / 2, 50, "rgba(255,102,153,1.0)");
            let maxOpacity = 1.0;  // 最大透明度
            let minOpacity = 0.0;  // 最小透明的都
            let state2 = 1;  // 0缩小 1放大
            let vOpacity = 0.0; // 自身透明度

            // ======== 处理小球颜色 =======
            let ball3 = new Ball(canvas.width / 4 * 3, canvas.height / 2, 50);
            ball3.fill(ctx);
            // 红绿蓝三原色
            let red = 255;
            let green = 0;
            let blue = 0;
            // 目标三原色
            let targetRed = 10;
            let targetGreen = 255;
            let targetBlue = 55;



            // 定义缓动系数
            const easing = 0.02;

            (function drawFrame() {
                window.requestAnimationFrame(drawFrame, canvas);
                // 清空画布
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                // ======== 处理小球半径 =======
                if(state1 == 1){
                    let value =  maxRadius - ball1.radius;
                    if(value < 0.01){
                        state1 = 0
                    }else{
                        vRadius = (value) * easing;
                    }
                }else{
                    let value =  ball1.radius - minRadius;
                    if(value < 0.01){
                        state1 = 1
                    }else{
                        vRadius = - (value) * easing;
                    }
                }
                ball1.radius += vRadius;
                ball1.fill(ctx);

                // ======== 处理小球透明度 =======
                if(state2 == 1){
                    let value =  maxOpacity - vOpacity;
                    if(value < 0.01){
                        state2 = 0
                    }else{
                        vOpacity += (value) * easing;
                    }
                }else{
                    let value =  vOpacity - minOpacity;
                    if(value <  0.01){
                        state2 = 1
                    }else{
                        vOpacity -= (value) * easing;
                    }
                }
                ball2.color = "rgba(255,102,153," + vOpacity + ")";
                ball2.fill(ctx);


                // ======== 处理小球颜色 =======
                let vRed = (targetRed - red) * easing;
                let vGreen = (targetGreen - green) * easing;
                let vBlue = (targetBlue - blue) * easing;

                red += vRed;
                green += vGreen;
                blue += vBlue;

                let color = "rgba(" + parseInt(red) + "," + parseInt(green) + "," + parseInt(blue) + "," + "1.0)";
                ball3.color = color;

   
                ball3.fill(ctx);

                if(vBlue < 0.01){
                     red = 255;
                     green = 0;
                     blue = 0;
                }

            })();
        </script>
    </body>
</html>

4.2 弹性动画

弹性动画，类似于我们上一节所讲的缓动动画，都是把一个物体从一个地方位移到另一个地方的效果。

弹性动画不同之处在于，物体运动到某个地方要再弹一会儿，类似于 弹簧。

弹性动画中，距离和加速度成正比，物体距离终点越远，加速度越大，物体接近终点后，加速度会很小，但是它依然在加速，然后物体会越过终点，反向加速度随之变大，会把物体拉回来，会使物体在终点处反弹一段时间，最终在摩擦力作用下停止运动。

4.2.1 基础案例

  • 代码
<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8" />
        <title>Document</title>
    </head>
    <body>
        <canvas id="myCanvas" width="600" height="400" style="background: #ccc"> </canvas>

        <script>
            class Ball{
                constructor(x = 0,y = 0,radius = 12,color = '#6699FF'){
                    //小球中心的x坐标
                    this.x = x;
                     //小球中心的y坐标
                    this.y = y;
                    //小球半径
                    this.radius = radius;
                    // 小球颜色
                    this.color = color;
                }

                fill (ctx){
                    ctx.fillStyle = this.color;
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.radius, 0, 360 * Math.PI / 180);
                    ctx.closePath();
                    ctx.fill()
                }
            }

            // 获取canvas对象
            const canvas = document.getElementById("myCanvas");
            // 获取上下文对象context
            const ctx = canvas.getContext("2d");

            // 初始化小球
            const ball = new Ball(0, canvas.height / 2,30);
            // 定义终点位置
            const targetX = canvas.width / 2;
            // 定义弹性系数 值越大弹得越快
            const spring = 0.005;
            // 速度
            let vx = 0;
            // 定义摩擦力
            const friction = 0.98;

            (function drawFrame() {
                window.requestAnimationFrame(drawFrame, canvas);
                // 清空画布
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                // 计算加速度
                const ax = (targetX - ball.x) * spring;
                // 加速度赋值给速度
                vx += ax ;
                // 没有摩檫力 会一直弹
                vx *= friction;
                ball.x += vx

                ball.fill(ctx)

               
                // 重新弹
                if(ball.x - targetX > -0.001 && ball.x - targetX < 0.001  ){
                    vx = 0;
                    ball.x = 0;
                }

            })();
        </script>
    </body>
</html>

4.2.3 悠悠球效果

<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8" />
        <title>Document</title>
    </head>
    <body>
        <canvas id="myCanvas" width="600" height="400" style="background: #ccc"> </canvas>

        <script>
            class Ball{
                constructor(x = 0,y = 0,radius = 12,color = '#6699FF'){
                    //小球中心的x坐标
                    this.x = x;
                     //小球中心的y坐标
                    this.y = y;
                    //小球半径
                    this.radius = radius;
                    // 小球颜色
                    this.color = color;
                }

                fill (ctx){
                    ctx.fillStyle = this.color;
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.radius, 0, 360 * Math.PI / 180);
                    ctx.closePath();
                    ctx.fill()
                }
            }

            // 定义鼠标事件
            const getMouse = (element) => {
                let mouse = { x: 0, y: 0 } // 存储鼠标位置信息
                element.addEventListener('mousemove', e => {
                    let x = e.pageX
                    let y = e.pageY
                    // 计算鼠标在canvas画布中的相对位置
                    mouse.x = x - element.offsetLeft
                    mouse.y = y - element.offsetTop
                })
                return mouse
            }

            // 获取canvas对象
            const canvas = document.getElementById("myCanvas");
            // 获取上下文对象context
            const ctx = canvas.getContext("2d");

            // 初始化小球
            const ball = new Ball(canvas.width / 2, canvas.height / 2,30);
            const mouse = getMouse(canvas);
            // 定义弹性系数 值越大弹得越快
            const spring = 0.01;
            // 定义速度
            let vx = 0;
            let vy = 0;
            // 定义摩擦力
            const friction = 0.95;
            // 定义重力
            const gravity = 1;

            (function drawFrame() {
                window.requestAnimationFrame(drawFrame, canvas);
                // 清空画布
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                const ax = (mouse.x - ball.x) * spring // 计算加速度
                const ay = (mouse.y - ball.y) * spring

                vx += ax
                vy += ay

                vy += gravity

                vx *= friction
                vy *= friction

                ball.x += vx
                ball.y += vy
                
                ball.fill(ctx)

                // 将指针与小球连线
                ctx.beginPath()
                ctx.moveTo(ball.x, ball.y)
                ctx.lineTo(mouse.x, mouse.y)
                ctx.stroke()
                ctx.closePath()

            })();
        </script>
    </body>
