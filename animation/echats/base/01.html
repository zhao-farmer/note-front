一、Echarts 图表使用

1.1 初始使用

1.1.1 使用步骤

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8" />
    <title>Document</title>
    <!--   步骤1：引入echats.js文件  -->
    <script src="../js/echarts-5.5.js"></script>

    <!-- 
      步骤1：引入echats.js文件
      步骤2：准备一个呈现图表的盒子
      步骤3：初始化echats实力对象
      步骤4：准备配置项
      步骤5：将配置项设置给echats实力对象
    -->
</head>
<body>
    <!-- 步骤2：准备一个呈现图表的盒子 -->
    <div style="width: 600px; height: 400px"></div>
    <script>
        // 步骤3：初始化echats实力对象
        // 参数，dom，决定图表最终呈现的位置
        var mCharts = echarts.init(document.querySelector("div"));
        // 步骤4：准备配置项
        var option = {
            xAxis: {
                type: "category",
                data: ["小明", "小红", "小王"],
            },
            yAxis: {
                type: "value",
            },
            series: [
                {
                    name: "语文",
                    type: "bar",
                    data: [70, 92, 87],
                },
            ],
        };
        // 步骤5：将配置项设置给echats实力对象
        mCharts.setOption(option);
    </script>
</body>
</html>

1.1.2 相关配置详解

  • 属性

    • xAxis: 直角坐标系中的 x 轴
    • yAxis: 直角坐标系中的 y 轴
    • series: 系列列表。每个系列通过 type 决定自己的图表类型

  • 代码

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8" />
    <title>Document</title>
    <script src="../js/echarts-5.5.js"></script>
</head>
<body>
    <div style="width: 600px; height: 400px"></div>
    <script>
        // 参数，dom，决定图表最终呈现的位置
        var mCharts = echarts.init(document.querySelector("div"));

        var option = {
            title: {
                text: "成绩",
                link: "https://www.baidu.com",
                textStyle: {
                    color: "red",
                },
            },
            xAxis: {
                type: "category", //类目轴
                data: ["小明", "小红", "小王"],
            },
            yAxis: {
                type: "value", // 数值轴

                {
                    name: "语文",
                    type: "bar", //柱状图 pie 扇形图 line 线形图
                    data: [70, 92, 87],

        };

    </script>
</body>

        // 标记：最大值、最小值、平均值
	    //      markPoint(最大值、最小值) markLine(平均值)
        // 显示：数值显示	柱宽度 横向柱状图

1.3.2 tooltip-提示框

  • tooltip：提示框组件，用于配置鼠标滑过或点击图表时的显示框

    • 触发类型：trigger
      • item、axis
    • 触发时机：triggerOn
      • mouserover、click
    • 格式化：formatter
      • 字符串模板、回调函数

  • 代码

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8" />
    <title>Document</title>
    <script src="../js/echarts-5.5.js"></script>
</head>
<body>
    <div style="width: 600px; height: 400px"></div>
    <script>
        var mCharts = echarts.init(document.querySelector("div"));

        var xDataArr = ["张三", "李四", "王五", "闰土", "小明", "茅台", "二妞", "大强"];
        var yDaataArr = [88, 92, 63, 77, 94, 80, 72, 86];
        var option = {
            title: {
                text: "成绩展示",
                textStyle: {
                    color: "red",
                },
                borderWidth: 5,
                borderColor: "blue",
                borderRadius: 5,
                left: 30,
                top: 10,
            },
            tooltip: {
                //trigger的值 item：移入柱状图中  axis:移入坐标轴中
                trigger: "axis",
                //triggerOn mouse 鼠标移入 click:点击
                triggerOn: "click",
                // formatter 提示框浮层内容格式器，支持字符串模板和回调函数两种形式。
                // formatter: "{b}的成绩是{c}",
                formatter: function (arg) {
                    return arg[0].name + "的分数是：" + arg[0].data;
                },
            },
            xAxis: {
                type: "category",
                data: xDataArr,
            },
            yAxis: {
                type: "value",
            },
            series: [
                {
                    name: "语文",
                    type: "bar",
                    data: yDaataArr,
                },
            ],
        };

        mCharts.setOption(option);
    </script>
</body>
</html>

        //    易用性，功能，拍照，跑分，续航，每个维度的最大值都是100

            {

                max: 100,
            },
            {
                name: "功能",
                max: 100,
            },
            {
                name: "拍照",

            },
            {
                name: "跑分",
                max: 100,

            {
                name: "续航",
                max: 100,
            },

        var option = {
            // 通过 radar属性配置

                indicator: dataMax,

            //  准备具体产品的数据

                {

                    data: [

                            name: "中兴手机1",

                        },
                    ],
                },
            ],
        };

        // 步骤四
        mCharts.setOption(option);
    </script>

1.11 总结

  1. 属性总结

  1. 使用场景
    • 柱状图:柱状图描述的是分类数据，呈现的是每一个分类中有多少
    • 折线图:折线图常用来分析数据随时间的变化趋势
    • 散点图:散点图可以帮助我们推断出不同维度数据之间的相关性
    • 饼图:饼图可以很好地帮助用户快速了解不同分类的数据的占比情况
    • 地图:地图主要可以帮助我们从宏观的角度快速看出不同地理位置上数据的差异
    • 雷达图:雷达图可以用来分析多个维度的数据与标准数据的对比情况
    • 仪表盘:仪表盘可以更直观的表现出某个指标的进度或实际情况
