# 三、四大组件

## 3.1 弹出窗口

1. 介绍 

    - 是什么？
        
        弹出窗口是用户点击浏览器工具栏中的扩展图标时显示的小窗口界面。

    - 主要特点

        - 临时性​​: 只在点击时显示，失去焦点时关闭
        - ​​轻量级​​: 通常是小型的交互界面
        - 快速访问​​: 提供最常用的功能入口

    - manifest.json 配置

        ```json
        {
            "action": {
                "default_popup": "popup.html",
                "default_title": "点击打开设置"
            }
        }
        ```


2. 代码示例

        ```json
        {
            "manifest_version": 3,
        ```

    - popup.js

        ```js
        class SimplePopup {
            constructor() {
                this.count = 0;
                this.init();
            }

            async init() {
                // 加载保存的数据
                await this.loadSavedData();

                // 设置事件监听器
                this.setupEventListeners();

                // 更新UI
                this.updateUI();

                this.setStatus("就绪");
            }

            // 设置事件监听器
            setupEventListeners() {
                // 改变背景颜色按钮
                document.getElementById("change-bg-color").addEventListener("click", () => {
                    this.changeBackgroundColor();
                });

                // 获取页面信息按钮
                document.getElementById("get-page-info").addEventListener("click", () => {
                    this.getPageInfo();
                });

                // 计数器按钮
                document.getElementById("increase").addEventListener("click", () => {
                    this.changeCount(1);
                });

                document.getElementById("decrease").addEventListener("click", () => {
                    this.changeCount(-1);
                });

                document.getElementById("reset").addEventListener("click", () => {
                    this.resetCount();
                });

                // 保存笔记按钮
                document.getElementById("save-note").addEventListener("click", () => {
                    this.saveNote();
                });
            }

            // 改变页面背景颜色
            async changeBackgroundColor() {
                this.setStatus("处理中...");

                try {
                    // 获取当前标签页
                    const [tab] = await chrome.tabs.query({ active: true, currentWindow: true });

                    // 在目标标签页中执行脚本
                    await chrome.scripting.executeScript({
                        target: { tabId: tab.id },
                        func: () => {
                            // 随机颜色
                            const colors = ["#ff6b6b", "#4ecdc4", "#45b7d1", "#f9ca24", "#6c5ce7"];
                            const randomColor = colors[Math.floor(Math.random() * colors.length)];
                            document.body.style.backgroundColor = randomColor;
                            document.body.style.transition = "background-color 0.5s ease";
                        },
                    });

                    this.setStatus("背景颜色已改变");
                } catch (error) {
                    console.error("改变背景颜色失败:", error);
                    this.setStatus("操作失败", "error");
                }
            }

            // 获取页面信息
            async getPageInfo() {
                this.setStatus("获取中...");

                try {
                    const [tab] = await chrome.tabs.query({ active: true, currentWindow: true });

                    await chrome.scripting.executeScript({
                        target: { tabId: tab.id },
                        func: () => {
                            // 获取页面信息
                            const pageInfo = {
                                title: document.title,
                                url: window.location.href,
                                wordCount: document.body.innerText.trim().split(/\s+/).length,
                                links: document.querySelectorAll("a").length,
                                images: document.querySelectorAll("img").length,
                            };

                            // 显示提示
                            alert(`页面信息:\n标题: ${pageInfo.title}\n字数: ${pageInfo.wordCount}\n链接: ${pageInfo.links}\n图片: ${pageInfo.images}`);
                        },
                    });

                    this.setStatus("页面信息已显示");
                } catch (error) {
                    console.error("获取页面信息失败:", error);
                    this.setStatus("操作失败", "error");
                }
            }

            // 改变计数器
            changeCount(delta) {
                this.count += delta;
                this.updateCount();
                this.saveData();
            }

            // 重置计数器
            resetCount() {
                this.count = 0;
                this.updateCount();
                this.saveData();
                this.setStatus("计数器已重置");
            }

            // 更新计数器显示
            updateCount() {
                document.getElementById("count").textContent = this.count;
            }

            // 保存笔记
            async saveNote() {
                const noteText = document.getElementById("notes").value.trim();

                if (!noteText) {
                    this.setStatus("请输入笔记内容", "warning");
                    return;
                }

                try {
                    // 保存到存储
                    await chrome.storage.local.set({
                        note: noteText,
                        noteSavedAt: new Date().toLocaleString(),
                    });

                    this.setStatus("笔记已保存");

                    // 清空输入框
                    document.getElementById("notes").value = "";
                } catch (error) {
                    console.error("保存笔记失败:", error);
                    this.setStatus("保存失败", "error");
                }
            }

            // 加载保存的数据
            async loadSavedData() {
                try {
                    const data = await chrome.storage.local.get(["count", "note"]);

                    // 加载计数器
                    if (data.count !== undefined) {
                        this.count = data.count;
                    }

                    // 加载笔记
                    if (data.note) {
                        document.getElementById("notes").value = data.note;
                    }
                } catch (error) {
                    console.error("加载数据失败:", error);
                }
            }

            // 保存数据
            async saveData() {
                try {
                    await chrome.storage.local.set({
                        count: this.count,
                    });
                } catch (error) {
                    console.error("保存数据失败:", error);
                }
            }

            // 设置状态消息
            setStatus(message, type = "success") {
                const statusElement = document.getElementById("status");
                statusElement.textContent = message;

                // 根据类型设置颜色
                const colors = {
                    success: "#4cd964",
                    error: "#ff3b30",
                    warning: "#ffcc00",
                    info: "#007bff",
                };

                statusElement.style.color = colors[type] || colors.success;
            }

            // 更新UI
            updateUI() {
                this.updateCount();
            }
        }

        // 初始化弹出窗口
        document.addEventListener("DOMContentLoaded", () => {
            const popup = new SimplePopup();

            // 暴露到全局用于调试
            window.popup = popup;
        });

        // 添加一些工具函数
        function showNotification(message, type = "info") {
            // 创建通知元素
            const notification = document.createElement("div");
            notification.style.cssText = `
            position: fixed;
            top: 10px;
            right: 10px;
            background: ${type === "error" ? "#ff3b30" : type === "success" ? "#4cd964" : "#007bff"};
            color: white;
            padding: 10px 15px;
            border-radius: 5px;
            z-index: 1000;
            font-size: 12px;
            animation: slideIn 0.3s ease;
        `;

            notification.textContent = message;
            document.body.appendChild(notification);

            // 3秒后移除
            setTimeout(() => {
                notification.remove();
            }, 3000);
        }

        // 添加CSS动画
        const style = document.createElement("style");
        style.textContent = `
        @keyframes slideIn {
            from {
            transform: translateX(100%);
            opacity: 0;
            }
            to {
            transform: translateX(0);
            opacity: 1;
            }
        }
        `;
        document.head.appendChild(style);
        ```

## 3.2 选项页面

1. 介绍

    - 是什么？
    
        选项页面是扩展的配置界面，用户可以通过它进行详细的设置。
    - 主要特点
        
        - 持久性​​: 独立页面，可以长时间打开
        - 完整性​​: 提供完整的配置选项
        - 复杂性​​: 可以包含复杂的表单和设置

    - manifest.json 配置

        ```json
        {
            "options_ui": {
                "page": "options.html",
                "open_in_tab": false
            }
        }
        ```
2. 代码示例

        ```json
        {
        ```

        ```js
                sendResponse({ success: true });
            }
        });

            console.log("应用设置:", settings);

        // 监听扩展安装
        chrome.runtime.onInstalled.addListener(() => {

                if (!result.extensionSettings) {

                        fontSize: 14,
                        saveInterval: 5,

            });
        ```

        - 中枢性​​: 协调各个组件之间的通信

        ```js
        // background.js - 简单后台服务
        class BackgroundService {
            constructor() {
                this.isEnabled = true;
                this.counter = 0;
                this.init();
            }

            async init() {
                console.log("🔧 后台服务初始化中...");

                // 加载保存的设置
                await this.loadSettings();

                // 设置事件监听器
                this.setupEventListeners();

                // 启动定时任务
                this.startScheduledTasks();

                console.log("✅ 后台服务初始化完成");
            }

            // 加载设置
            async loadSettings() {
                try {
                    const result = await chrome.storage.local.get(["settings"]);
                    if (result.settings) {
                        this.isEnabled = result.settings.enabled ?? true;
                        this.counter = result.settings.counter ?? 0;
                    }
                    console.log("📁 设置加载完成:", { isEnabled: this.isEnabled, counter: this.counter });
                } catch (error) {
                    console.error("❌ 加载设置失败:", error);
                }
            }

            // 保存设置
            async saveSettings() {
                try {
                    await chrome.storage.local.set({
                        settings: {
                            enabled: this.isEnabled,
                            counter: this.counter,
                            lastUpdated: new Date().toISOString(),
                        },
                    });
                    console.log("💾 设置已保存");
                } catch (error) {
                    console.error("❌ 保存设置失败:", error);
                }
            }

            // 设置事件监听器
            setupEventListeners() {
                // 扩展安装事件
                chrome.runtime.onInstalled.addListener(details => {
                    this.handleInstallation(details);
                });

                // 消息监听
                chrome.runtime.onMessage.addListener((request, sender, sendResponse) => {
                    return this.handleMessage(request, sender, sendResponse);
                });

                // 警报监听
                chrome.alarms.onAlarm.addListener(alarm => {
                    this.handleAlarm(alarm);
                });

                // 存储变化监听
                chrome.storage.onChanged.addListener((changes, area) => {
                    this.handleStorageChange(changes, area);
                });

                console.log("👂 事件监听器设置完成");
            }

            // 处理扩展安装
            handleInstallation(details) {
                console.log("📦 扩展安装事件:", details.reason);

                if (details.reason === "install") {
                    this.showNotification("欢迎使用！", "扩展已成功安装。");
                    this.initializeDefaultSettings();
                } else if (details.reason === "update") {
                    console.log("🔄 扩展已更新到新版本");
                }
            }

            // 初始化默认设置
            async initializeDefaultSettings() {
                const defaultSettings = {
                    enabled: true,
                    counter: 0,
                    installedAt: new Date().toISOString(),
                };

                await chrome.storage.local.set({ settings: defaultSettings });
                this.isEnabled = true;
                this.counter = 0;

                console.log("⚙️ 默认设置已初始化");
            }

            // 处理消息
            handleMessage(request, sender, sendResponse) {
                console.log("📨 收到消息:", request, "来自:", sender);

                switch (request.type) {
                    case "GET_STATUS":
                        this.handleGetStatus(sendResponse);
                        return true; // 保持消息通道开放

                    case "TOGGLE_SERVICE":
                        this.handleToggleService(request.data, sendResponse);
                        return true;

                    case "INCREMENT_COUNTER":
                        this.handleIncrementCounter(request.data, sendResponse);
                        return true;

                    case "GET_STATS":
                        this.handleGetStats(sendResponse);
                        return true;

                    case "SHOW_NOTIFICATION":
                        this.handleShowNotification(request.data, sendResponse);
                        return true;

                    default:
                        console.warn("⚠️ 未知消息类型:", request.type);
                        sendResponse({ success: false, error: "未知消息类型" });
                }
            }

            // 处理获取状态
            async handleGetStatus(sendResponse) {
                try {
                    const result = await chrome.storage.local.get(["settings"]);
                    sendResponse({
                        success: true,
                        data: {
                            enabled: this.isEnabled,
                            counter: this.counter,
                            lastUpdated: result.settings?.lastUpdated,
                        },
                    });
                } catch (error) {
                    sendResponse({ success: false, error: error.message });
                }
            }

            // 处理切换服务
            async handleToggleService(data, sendResponse) {
                this.isEnabled = data.enabled ?? !this.isEnabled;

                await this.saveSettings();

                // 根据状态启动或停止任务
                if (this.isEnabled) {
                    this.startScheduledTasks();
                } else {
                    this.stopScheduledTasks();
                }

                sendResponse({
                    success: true,
                    data: { enabled: this.isEnabled },
                });

                this.showNotification("服务状态已更新", this.isEnabled ? "服务已启用" : "服务已禁用");
            }

            // 处理增加计数器
            async handleIncrementCounter(data, sendResponse) {
                const amount = data.amount || 1;
                this.counter += amount;

                await this.saveSettings();

                sendResponse({
                    success: true,
                    data: { counter: this.counter },
                });

                console.log("🔢 计数器增加:", amount, "当前值:", this.counter);
            }

            // 处理获取统计信息
            async handleGetStats(sendResponse) {
                try {
                    const result = await chrome.storage.local.get(["settings"]);
                    const settings = result.settings || {};

                    sendResponse({
                        success: true,
                        data: {
                            enabled: this.isEnabled,
                            counter: this.counter,
                            installedAt: settings.installedAt,
                            lastUpdated: settings.lastUpdated,
                            serviceUptime: this.getServiceUptime(settings.installedAt),
                        },
                    });
                } catch (error) {
                    sendResponse({ success: false, error: error.message });
                }
            }

            // 处理显示通知
            handleShowNotification(data, sendResponse) {
                this.showNotification(data.title, data.message, data.iconUrl);
                sendResponse({ success: true });
            }

            // 处理警报
            handleAlarm(alarm) {
                console.log("⏰ 警报触发:", alarm.name);

                switch (alarm.name) {
                    case "heartbeat":
                        this.handleHeartbeat();
                        break;

                    case "daily-reminder":
                        this.handleDailyReminder();
                        break;

                    case "cleanup":
                        this.handleCleanup();
                        break;
                }
            }

            // 处理心跳检测
            handleHeartbeat() {
                if (!this.isEnabled) return;

                console.log("💓 心跳检测 - 服务运行正常");
                this.counter++;
                this.saveSettings();
            }

            // 处理每日提醒
            handleDailyReminder() {
                if (!this.isEnabled) return;

                this.showNotification("每日提醒", `今天是使用扩展的第 ${Math.ceil((new Date() - new Date(this.getInstallationDate())) / (1000 * 60 * 60 * 24))} 天！`, "icons/icon48.png");
            }

            // 处理清理任务
            async handleCleanup() {
                console.log("🧹 执行清理任务");

                // 这里可以添加清理逻辑，比如清理过期的缓存数据
                const result = await chrome.storage.local.get(null);
                console.log("当前存储数据:", result);
            }

            // 处理存储变化
            handleStorageChange(changes, area) {
                if (area === "local" && changes.settings) {
                    console.log("📦 存储设置已更改");

                    const newSettings = changes.settings.newValue;
                    if (newSettings) {
                        this.isEnabled = newSettings.enabled ?? this.isEnabled;
                        this.counter = newSettings.counter ?? this.counter;
                    }
                }
            }

            // 启动定时任务
            startScheduledTasks() {
                console.log("🚀 启动定时任务");

                // 创建心跳检测警报（每分钟一次）
                chrome.alarms.create("heartbeat", {
                    periodInMinutes: 1,
                });

                // 创建每日提醒（每天一次）
                chrome.alarms.create("daily-reminder", {
                    periodInMinutes: 60 * 24, // 24小时
                });

                // 创建清理任务（每6小时一次）
                chrome.alarms.create("cleanup", {
                    periodInMinutes: 60 * 6, // 6小时
                });
            }

            // 停止定时任务
            stopScheduledTasks() {
                console.log("🛑 停止定时任务");

                chrome.alarms.clearAll();
            }

            // 显示通知
            showNotification(title, message, iconUrl = "icons/icon48.png") {
                if (!this.isEnabled) return;

                chrome.notifications.create({
                    type: "basic",
                    iconUrl: chrome.runtime.getURL(iconUrl),
                    title: title,
                    message: message,
                    priority: 1,
                });

                console.log("📢 通知已发送:", title);
            }

            // 获取服务运行时间
            getServiceUptime(installedAt) {
                if (!installedAt) return "未知";

                const installedDate = new Date(installedAt);
                const now = new Date();
                const diff = now - installedDate;

                const days = Math.floor(diff / (1000 * 60 * 60 * 24));
                const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

                return `${days}天 ${hours}小时`;
            }

            // 获取安装日期
            getInstallationDate() {
                return new Date().toISOString().split("T")[0];
            }

            // 工具方法：延迟执行
            sleep(ms) {
                return new Promise(resolve => setTimeout(resolve, ms));
            }

            // 工具方法：重试操作
            async retryOperation(operation, maxRetries = 3, delay = 1000) {
                for (let attempt = 1; attempt <= maxRetries; attempt++) {
                    try {
                        return await operation();
                    } catch (error) {
                        if (attempt === maxRetries) throw error;
                        console.warn(`操作失败，第${attempt}次重试...`);
        ```

## 3.4 内容脚本


1. 介绍

    - 是什么？
        
        内容脚本是注入到网页上下文中的脚本，可以直接操作页面DOM。
    - 主要特点
        
        - ​隔离性​​: 在独立环境中运行，与页面JS隔离
        - OM访问​​: 可以直接访问和修改页面DOM
        - 受限性​​: 不能访问大部分Chrome API
    - manifest.json 配置

        ```json
        {
            "content_scripts": [
                {
                    "matches": ["https://*.example.com/*"],
                    "js": ["content.js"],
                    "css": ["content.css"],
                    "run_at": "document_idle"
                }
            ]
        }
        ```

        ```js
            constructor() {
                this.isEnabled = true;
                this.pageStats = {
                    wordCount: 0,
                    linkCount: 0,
                    headingCount: 0,
                    imageCount: 0,
                };

                this.init();

            async init() {
                console.log("🚀 内容脚本初始化 - 当前页面:", window.location.href);

                // 等待页面完全加载
                if (document.readyState === "loading") {
                    document.addEventListener("DOMContentLoaded", () => this.onPageLoad());
                } else {
                    this.onPageLoad();
                }

                // 设置消息监听
                this.setupMessageListeners();

                // 设置突变观察器
                this.setupMutationObserver();
            }

            onPageLoad() {
                console.log("📄 页面加载完成:", document.title);

                // 收集页面统计信息
                this.collectPageStats();

                // 注入自定义UI
                this.injectCustomUI();

                // 根据页面类型执行特定操作
                this.handleSpecificSites();

                // 发送页面分析完成消息
                this.sendPageAnalysis();

            // 收集页面统计信息
            collectPageStats() {
                this.pageStats = {
                    wordCount: this.countWords(),
                    linkCount: this.countLinks(),
                    headingCount: this.countHeadings(),
                    imageCount: this.countImages(),
                    readingTime: this.calculateReadingTime(),
                    pageSize: this.calculatePageSize(),
                    timestamp: new Date().toISOString(),
                };

                console.log("📊 页面统计:", this.pageStats);
            }

            // 统计字数
            countWords() {
                const text = document.body.textContent || "";
                return text
                    .trim()
                    .split(/\s+/)
                    .filter(word => word.length > 0).length;
            }

            // 统计链接
            countLinks() {
                return document.querySelectorAll("a[href]").length;
            }

            // 统计标题
            countHeadings() {
                let count = 0;
                for (let i = 1; i <= 6; i++) {
                    count += document.querySelectorAll(`h${i}`).length;
                }
                return count;
            }

            // 统计图片
            countImages() {
                return document.querySelectorAll("img").length;
            }

            // 计算阅读时间
            calculateReadingTime() {
                const wordsPerMinute = 200;
                return Math.ceil(this.pageStats.wordCount / wordsPerMinute);
            }

            calculatePageSize() {
                const html = document.documentElement.outerHTML;
                return new Blob([html]).size;

                // 创建切换按钮
                this.createToggleButton();

                console.log("🎨 自定义UI已注入");

                document.body.appendChild(sidebar);

                sidebar.querySelector("#extension-remove").addEventListener("click", () => {

            createToggleButton() {
                const toggleBtn = document.createElement("button");
                toggleBtn.className = "extension-toggle";
                toggleBtn.textContent = "⚡";
                toggleBtn.title = "切换扩展功能";

            highlightHeadings() {

            showTooltip(element, text) {

            removeTooltip() {
                const tooltip = document.querySelector(".extension-tooltip");
                if (tooltip) {
                    tooltip.remove();
                }
            }

            // 移除高亮
            removeHighlights() {
                const highlighted = document.querySelectorAll(".extension-highlight");
                highlighted.forEach(element => {
                    element.classList.remove("extension-highlight");
                });

                this.removeTooltip();
                console.log("🗑️ 已清除高亮");
            }

            // 移除已存在的UI
            removeExistingUI() {
                const existingUI = document.querySelectorAll(".extension-sidebar, .extension-toggle");
                existingUI.forEach(element => element.remove());
            }

            // 处理特定网站
            handleSpecificSites() {
                const url = window.location.href;

                if (url.includes("github.com")) {
                    this.enhanceGitHub();
                } else if (url.includes("stackoverflow.com")) {
                    this.enhanceStackOverflow();

            // 设置消息监听
            setupMessageListeners() {
                chrome.runtime.onMessage.addListener((request, sender, sendResponse) => {
                    console.log("📨 收到消息:", request);

                    switch (request.action) {
                        case "GET_PAGE_STATS":
                            sendResponse({ success: true, stats: this.pageStats });
                            break;

                        case "HIGHLIGHT_HEADINGS":
                            this.highlightHeadings();
                            sendResponse({ success: true });
                            break;

                        case "TOGGLE_EXTENSION":
                            this.isEnabled = request.enabled;

                            sendResponse({ success: true, enabled: this.isEnabled });
                            break;

                        case "EXPORT_STATS":
                            this.exportStats();
                            sendResponse({ success: true });
                            break;

                        default:
                            sendResponse({ success: false, error: "未知操作" });

            setupMutationObserver() {
                const observer = new MutationObserver(mutations => {
                    mutations.forEach(mutation => {
                        if (mutation.addedNodes.length > 0) {
                            this.handleDynamicContent(mutation.addedNodes);
                        }
                    });
                });

                observer.observe(document.body, {
                    childList: true,
                    subtree: true,
                });

            handleDynamicContent(nodes) {

            exportStats() {
                const dataStr = JSON.stringify(this.pageStats, null, 2);

                    clearTimeout(timeout);

                };
                clearTimeout(timeout);
                timeout = setTimeout(later, wait);
        ```
