四、ES9

4.1 …(三点)的使用

4.1.1 rest(对象剩余参数)

let obj = {
    name:'kerwin',
    age:100,
    location:"dalian"
}

// 1. 解构赋值使用
let {name,...other} = obj

console.log(name,other);  
// kerwin  {age: 100, location: 'dalian'}

// 2. 在函数的参数中使用
function test({name,...other}){
    console.log(name,other);
}

test(obj)
// kerwin  {age: 100, location: 'dalian'}

4.1.2 扩展运算符

// 1.数据合并
let obj1 = {
    name:'tiechui',
    location:"dalian"
}
let obj2 ={
    name:'xiaoming',
    age:18
}
// 类似 Object.assign
// 不同点：Object.assign 第一个参数有的不参与替换
let obj3 = {...obj1,...obj2}
console.log(obj3);
// {name: 'xiaoming', location: 'dalian', age: 18}
// 案例
function ajax(options){
    const defaultOptions = {
        methods:"get",
        async:true
    }
    options = {...defaultOptions,...options}
    console.log(options);   
}
ajax({
    url:"",
    methods:"post",
})
// {methods: 'post', async: true, url: ''}
ajax({
    url:"",
})
// {methods: 'post', async: true, url: ''}


// 2. 浅复制（使用扩展运算符 ）
let obj5 = {
    name:"kerwin",
    age:100
}
let obj6 = {...obj5}
obj6.age = 10
console.log(obj5);
// {name: 'kerwin', age: 100}

4.2 正则扩展-分组命名

let str = "今天是2022-10-10"
let reg = /[0-9]{4}-[0-9]{2}-[0-9]{2}/

// exec 基础捕获
console.log(reg.exec(str));
//['2022-10-10', index: 3, input: '今天是2022-10-10', groups: undefined]

// 分组 痛点：根据索引去取值，如果结构改变 会更改更多
let reg2 = /([0-9]{4})-([0-9]{2})-([0-9]{2})/
console.log(reg2.exec(str));    
//  ['2022-10-10', '2022', '10', '10', index: 3, input: '今天是2022-10-10', groups: undefined]

// ES9 分组加对象
let reg3 = /(?<year>[0-9]{4})-(?<mouth>[0-9]{2})-(?<day>[0-9]{2})/
let res = reg3.exec(str)
let {year,mouth,day} = res.groups
console.log(year,mouth,day);    //2022 10 10

4.3 Promise.finally()

无论是成功还是失败, 都运行同样的代码, 比如隐藏对话框, 关闭数据连接

function ajax(){
    return new Promise((resolve,reject)=>{
        // resolve("data-111111")
        reject("err-111111")
    })
}

// 显示loading
ajax().then((data)=>{
    console.log(data);
}).catch(err=>{
    console.log("err",err);
}).finally(()=>{
    // 隐藏loading
    console.log("finally");
})

4.4 异步迭代

4.4.1 for循环中使用await等待Promise问题

/* 
    目标：异步任务按照顺序进行遍历
*/
function timer(t) {
    return new Promise(resolve => {
        setTimeout(() => {
            resolve("data-" + t);
        }, t);
    });
}
// 这种是并发任务
async function test() {
    let arr = [timer(1000), timer(2000), timer(3000)];
    /* 
        每次都是按照for循环的顺序执行 
        每次都是卡在await那里    
        如果第一个参数大于后面的 打印出的时间都是一样的
    */
    for (const item of arr) {
        console.log("start-", Date.now());
        console.log(await item);
        console.log("end-", Date.now());
    }
}
test();

4.4.2 生成器种使用Promise对象


/* 前期测试 - 数组同步遍历器  */

let arr = [1,2,3]
// 变为迭代器
let i = arr[Symbol.iterator]()
// 第一种： 只用迭代器的next方法
console.log(i.next());
console.log(i.next());
console.log(i.next());
console.log(i.next());
// 第二种：使用forof 遍历
for (const i of arr) {
    console.log(i); 
}

/* 前期测试 - 同步生成器 */
function *gen(){
    yield 111
    yield 222
}
let g = gen()
console.log(g.next());
console.log(g.next());
console.log(g.next());


// 如果生成器调度的是异步Promise 使用会反直觉
function ajax(data){
    return new Promise(resolve =>{
        resolve(data)
    })
}
// 异步生成器
function *gen2(){
    yield ajax(111)
    yield ajax(222)
}
let g2 = gen2() // g就是一个异步遍历器


// 只能使用这种方式 处理数据
g2.next().value.then(res=>{
    console.log(res);
})
g2.next().value.then(res=>{
    console.log(res);
})

4.4.3 解决方案（含有for await）

function ajax(data){
    return new Promise(resolve =>{
        resolve(data)
    })
}

// 解决方法1：使用链式调用
// 异步生成器 
async function *gen1(){
    yield ajax(111)
    yield ajax(222)
}
let g1 = gen1() // g就是一个异步遍历器
g1.next().then(res =>{
    console.log("g1",res);  
    return g1.next()
}).then(res=>{
    console.log("g1",res); 
    return g1.next()
}).then(res=>{
    console.log("g1",res);     
})

// 解决方法2： 配合 async await使用
// 由于返回的Promise对象 可以使用 await进行顺序调用
async function *gen2(){
    yield ajax(333)
    yield ajax(444)
}
let g2 = gen2()
async function test(){
    let res1 = await g2.next()
    console.log("g2",res1);   
    let res2 = await g2.next()
    console.log("g2",res2);  
    let res3 = await g2.next()
    console.log("g2",res3);  
}
test()

// 解决方法3：ES9中的for循环加await
// 比较简便
async function *gen3(){
    yield ajax(555)
    yield ajax(666)
}
let g3 = gen3()
async function test2(params) {
    let list = [g3.next(),g3.next(),g3.next()]
    for await(const item of list) {
        console.log(item);
    }
}
test2()

4.4.4 案例改造

/* 
    任务：异步任务按照顺序进行遍历
*/

function timer(t) {
    return new Promise(resolve => {
        setTimeout(() => {
            resolve("data-" + t);
        }, t);
    });
}

// 异步生成器
async function* gen() {
    yield timer(1000);
    yield timer(2000);
    yield timer(3000);
}
async function test() {
    // 异步遍历器数组
    let arr = [g.next(), g.next(), g.next()];
    // for await遍历
    for await (const item of arr) {
        console.log("start-", Date.now());
        console.log(await item);
        console.log("end-", Date.now());
    }
}
test();
