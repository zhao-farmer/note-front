<!DOCTYPE html>
<html lang="en">
  <head>

    <title>Document</title>
    <style>
      * {
        margin: 0;
        padding: 0;
      }
      img {
        vertical-align: top;
        width: 640px;
      }
      ul {
        list-style: none;
      }
      .outer {
        width: 640px;
        height: 390px;
        margin: 100px auto;
        position: relative;
      }
      .img-list {
        height: 100%;
      }
      .img-list li {
        position: absolute;
        opacity: 0;
        transition: all 0.7s;
      }
      li.current {
        z-index: 1;
        opacity: 1;
      }
      /* 
        设置两个箭头的样式
      */
      .prev-next a {
        font-size: 60px;
        color: #fff;
        font-weight: bold;
        text-decoration: none;
        position: absolute;
        height: 60px;
        top: 0;
        bottom: 0;
        margin: auto;
        z-index: 2;
        opacity: 0.5;
      }
      .prev-next a:hover {
        opacity: 1;
      }
      #next {
        right: 0;
      }
      /* 
        设置导航点的样式
      */
      .dot {
        position: absolute;
        z-index: 3;
        left: 0;
        right: 0;
        margin: auto;
        display: flex;
        justify-content: center;
        bottom: 5px;
      }
      .dot a {
        float: left;
        width: 20px;
        height: 20px;
        margin: 5px;
        border-radius: 50%;
        background-color: #fff;
        opacity: 0.5;
      }
      .dot a:hover,
      .dot .active {
        opacity: 1;
      }
    </style>
  </head>
  <body>
    <div class="outer">
      <ul class="img-list">
        <li class="current">
          <a href="#">
            <img src="./images/1.jpg" />
          </a>
        </li>
        <li>
          <a href="#">
            <img src="./images/2.jpg" />
          </a>
        </li>
        <li>
          <a href="#">
            <img src="./images/3.jpg" />
          </a>
        </li>
        <li>
          <a href="#">
            <img src="./images/4.jpg" />
          </a>
        </li>
        <li>
          <a href="#">
            <img src="./images/5.jpg" />
          </a>
        </li>
      </ul>
      <!-- 添加切换按钮 -->
      <div class="prev-next">
        <a id="prev" href="javascript:;"><</a>
        <a id="next" href="javascript:;">></a>
      </div>
      <!-- 添加导航点 -->
      <div class="dot">
        <a class="active" href="javascript:;"></a>
        <a href="javascript:;"></a>
        <a href="javascript:;"></a>
        <a href="javascript:;"></a>
        <a href="javascript:;"></a>
      </div>
    </div>
    <script>
      /* 
            自动切换图片
        */
      const toggleChange = (function () {
        let timer;
        return function () {
          if (timer == null) {
            timer = setTimeout(function auto() {
              changeImg("next");
              timer = setTimeout(auto, 2000);
            }, 2000);
          } else {
            clearTimeout(timer);
            timer = null;
          }
        };
      })();
      toggleChange();
      const outer = document.querySelector(".outer");
      outer.onmouseenter = () => {
        console.log("进入");
        toggleChange();
      };
      outer.onmouseleave = () => {
        console.log("离开");
        toggleChange();
      };
      /* 
          点击按钮 切换图片
      */
      const prevBtn = document.getElementById("prev");
      const nextBtn = document.getElementById("next");
      //所有的点
      const dots = Array.from(document.querySelectorAll(".dot a"));
      //图片的数组
      const imgArr = Array.from(document.querySelectorAll(".img-list li"));
      prevBtn.onclick = () => {
        changeImg("prev");
      };
      nextBtn.onclick = () => {
        changeImg("next");
      };
      /* 
        changeImg 用来切换图片
          参数：
            dir 切换图片的方向
              next
              prev
      
      */
      function changeImg(dir) {
        // 获取当前图片
        const current = document.querySelector(".img-list .current");
        //获取当前要显示图片的索引
        const index = imgArr.indexOf(current);
        let next;
        if (dir == "next") {
          next = current.nextElementSibling || imgArr[0];
        } else if (dir == "prev") {
          next = current.previousElementSibling || imgArr.at(-1);
        } else {
          next = imgArr[dir];
        }
        // 切换显示状态
        current.classList.remove("current");
        next.classList.add("current");
        //切换active
        const currentActive = document.querySelector(".active");
        currentActive.classList.remove("active");
        dots[index].classList.add("active");
      }
      /* 
        获取5个小点
      
      */
      document.addEventListener("click", (event) => {
        const index = dots.indexOf(event.target);
        if (index !== -1) {
          changeImg(index);
        }
      });
    </script>
  </body>
</html>

14.2 游戏贪吃蛇

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <title>Document</title>
    <style>
      * {
        margin: 0;
        padding: 0;
        box-sizing: border-box;
      }
      #main {
        height: 460px;
        width: 364px;
        background-color: #b7d4a8;
        border: 10px solid #000;
        border-radius: 20px;
        margin: 50px auto;
      }
      #stage {
        width: 304px;
        height: 304px;
        border: 2px solid #000;
        margin: 20px auto;
        position: relative;
      }
      #info {
        width: 304px;
        margin: 30px auto;
        display: flex;
        justify-content: space-between;
        font: bold 20px courier;
      }
      #food {
        width: 10px;
        height: 10px;
        position: absolute;
        top: 100px;
        left: 120px;
        display: flex;
        flex-wrap: wrap;
      }
      #food > div {
        width: 5px;
        height: 5px;
        background-color: #000;
        transform: rotate(45deg);
      }
      #snake div {
        width: 10px;
        height: 10px;
        background-color: #000;
        position: absolute;
        border: 1px solid #b7d4a8;
      }
    </style>
  </head>
  <body>
    <div id="main">
      <div id="stage">
        <div id="snake"><div></div></div>
        <div id="food">
          <div></div>
          <div></div>
          <div></div>
          <div></div>
        </div>
      </div>
      <div id="info">
        <div>SCORE: <span id="score">0</span></div>
        <div>LEVEL: <span id="level">1</span></div>
      </div>
    </div>
    <script>
      /* 
        1.禁止掉头
        2.游戏中止
        3.记分
      */
      // 获取蛇的容器
      const snake = document.getElementById("snake");
      // 获取蛇的各个部分
      let snakes = snake.getElementsByTagName("div");
      //获取食物
      const food = document.getElementById("food");
      // 获取分数和等级的span
      const scoreSpan = document.getElementById("score");
      const levelSpan = document.getElementById("level");
      //创建变量存储分数和等级
      let score = 0;
      let level = 0;
      /* 
          食物的坐标应该在0-290之间
          必须是整10的倍数
      */
      function changeFood() {
        // 生成0-29之间的随机数
        const x = Math.floor(Math.random() * 30);
        const y = Math.floor(Math.random() * 30);
        console.log(x, y);
        //设置食物的坐标
        food.style.left = x * 10 + "px";
        food.style.top = y * 10 + "px";
      }
      //定义一个变量用来存储蛇移动的方向
      let dir = "";
      /*
        绑定按键事件keydown keyup
          - 键盘事件只能绑定给可以获取焦点的元素或者是document
      */
      const keyArr = ["ArrowUp", "ArrowDown", "ArrowLeft", "ArrowRight"];
      //创建一个对象
      const reObj = {
        ArrowUp: "ArrowDown",
        ArrowDown: "ArrowUp",
        ArrowLeft: "ArrowRight",
        ArrowRight: "ArrowLeft",
      };
      /* 
        游戏禁止掉头
          构成的要件
            1.身体超过2
            2.不能时相反的方向
          处理：
            保持原来的方向不变（不修改dir的值）
      */
      //设置 keyActive 防止按键按的过快导致的问题 如：先按左键 立马按上键
      let keyActive = true;
      document.addEventListener("keydown", (event) => {
        if (keyActive && keyArr.includes(event.key)) {
          if (snakes.length < 2 || reObj[dir] !== event.key) {
            // 设置方向
            dir = event.key;
            keyActive = false;
          }
        }
      });
      /* 
        要使得身体可以和头一起移动，只需要在蛇移动时，变化蛇尾巴的位置
      
      */
      setTimeout(function move() {
        //获取蛇头
        const head = snakes[0];
        //获取蛇头的坐标
        let x = head.offsetLeft;
        let y = head.offsetTop;
        switch (dir) {
          case "ArrowUp":
            y -= 10;
            break;
          case "ArrowDown":
            y += 10;
            break;
          case "ArrowLeft":
            x -= 10;
            break;
          case "ArrowRight":
            //向右移动蛇
            x += 10;
            break;
        }
        if (head.offsetTop === food.offsetTop && head.offsetLeft === food.offsetLeft) {
          console.log("吃到食物了");
          //1.改变食物的位置
          changeFood();
          //2.给蛇添加身体
          snake.insertAdjacentHTML("beforeend", "<div/>");
          score++;
          scoreSpan.textContent = score;
          //检测等级
          if (score % 5 === 0 && level < 14) {
            level++;
            levelSpan.textContent = level + 1;
          }
        }
        /* 
          判断游戏是否结束
            1.撞墙
            2.撞自己
        
        */
        //判断是否撞墙
        if (x < 0 || x > 290 || y < 0 || y > 290) {
          alert("撞墙了！游戏结束！");
          //游戏结束 return 直接跳出不再执行下面代码
          return;
        }
        //判断是否撞到自己
        // snakes.length - 1 不判断尾巴与头的碰撞 这种情况应该放过
        if (dir) {
          for (let i = 0; i < snakes.length - 1; i++) {
            if (snakes[i].offsetLeft === x && snakes[i].offsetTop === y) {
              alert("撞到自己了，游戏结束！");
              return;
            }
          }
        }
        // 将最后一个元素移动到最开始的位置 完成移动
        //获取尾巴
        const tail = snakes[snakes.length - 1];
        //移动蛇的位置
        tail.style.left = x + "px";
        tail.style.top = y + "px";
        // 将尾巴移动到蛇头的位置
        snake.insertAdjacentElement("afterbegin", tail);
        //防止快速点击
        keyActive = true;
        setTimeout(move, 300 - level * 20);
      }, 300);
    </script>
  </body>
</html>
